function printNumbersWithDelays() {

            function printNumber() {
              if (number <= 7) {
                console.log(number);
                number++;
                setTimeout(printNumber, number * 1000); // Delay in milliseconds
              }
            }
          
            printNumber();
          }

Callback hell, also known as "Pyramid of Doom" or "Callback Pyramid," is a situation that arises in JavaScript when you have multiple nested callback functions. 

function stepOne(callback) {
            // ...
            callback(function () {
              // ...
              callback(function () {
                // ...
                callback(function () {
                  // ...
                });
              });
            });
          }
          

Q3:- Write the promise function to print numbers from 1 to 7, in which 1 should be printed after 1 sec, 2 should be printed after 2 sec, 3 should be printed after 3 sec, and so on. 

            Numbers
            1
            2

function printNumbersWithDelay() {
                let number = 1;
              
                function printNumber() {
                  return new Promise((resolve) => {
                    setTimeout(() => {
                      console.log(number);
                      resolve();
                    }, number * 1000); // Delay in milliseconds, multiplying by 1000 to convert to seconds
                  });
                }
              
                async function printSequentially() {
                  while (number <= 7) {
                    await printNumber();
                    number++;
                  }
                }
              
                printSequentially();
              }
              
              printNumbersWithDelay();

Q4:- Create a promise function accepting an argument, if yes is passed to the function then it should go to resolved state and print Promise Resolved, and if nothing is passed then it should go to reject the state and catch the error and print Promise Rejected 

function myPromise(argument) {
        return new Promise((resolve, reject) => {
          if (argument === 'yes') {
            resolve('Promise Resolved');
          } else {
            reject('Promise Rejected');
          }
        });
      }
      
      // Usage example:
      myPromise('yes')
        .then((result) => {
          console.log(result); 
        })
        .catch((error) => {
          console.error(error); 
        });
      
      myPromise()
        .then((result) => {
          console.log(result); 
        })
        .catch((error) => {
          console.error(error); 
        });
